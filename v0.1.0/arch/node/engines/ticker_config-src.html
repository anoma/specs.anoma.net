type TickerCfg :=
  mkTickerCfg@{
    example : Nat;
    value : String;
  };

module ticker_config_example;
  
  tickerCfg : EngineCfg TickerCfg :=
    mkEngineCfg@{
      node := Curve25519PubKey "0xabcd1234";
      name := "ticker";
      cfg :=
        mkTickerCfg@{
          example := 1;
          value := "hello world";
        };
    };
end;
