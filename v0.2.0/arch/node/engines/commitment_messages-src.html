module arch.node.engines.commitment_messages;

import prelude open;
import arch.node.types.identities open;

type RequestCommitment :=
  mkRequestCommitment@{
    data : Signable;
  };

type ReplyCommitment :=
  mkReplyCommitment@{
    commitment : Commitment;
    err : Option String;
  };

type CommitmentMsg :=
  | Request RequestCommitment
  | Reply ReplyCommitment;
