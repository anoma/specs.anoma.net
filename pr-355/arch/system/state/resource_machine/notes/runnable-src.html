trait
type Runnable KVSKey KVSDatum Executable ProgramState :=
  mkRunnable@{
    executeStep
      : Executable
        -> ProgramState
        -> Pair KVSKey KVSDatum
        -> Result

          (Pair ProgramState (List (Either KVSKey (Pair KVSKey KVSDatum))));
    halted : ProgramState -> Bool;
    startingState : ProgramState;
  };
