module arch.node.types.basics;

import prelude open public;

syntax alias Hash := Nat;
syntax alias RelTime := Nat;
syntax alias AbsTime := Nat;

Time : Type := Either RelTime AbsTime;

type Version : Type :=
  mkVersion@{
    major : Nat;
    minor : Nat;
    patch : Nat;
  };
