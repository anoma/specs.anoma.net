module node_architecture.engines.template_messages;

import prelude open;

syntax alias MethodOneArgOne := Nat;
syntax alias MethodOneArgTwo := Nat;
syntax alias MethodOneArgThree := Nat;
syntax alias MethodTwoArgOne := Nat;

type TemplateMsg :=
  | -- --8<-- [start:MethodOneMsg]
    MethodOneMsg {
      argOne : MethodOneArgOne;
      argTwo : MethodOneArgTwo;
      argThree : MethodOneArgThree
    }
  -- --8<-- [end:MethodOneMsg]
  | -- --8<-- [start:MethodTwoMsg]
    MethodTwoMsg {argOne : MethodTwoArgOne};

-- --8<-- [end:MethodTwoMsg]
module example-message-one;
  example_message_one : TemplateMsg :=
    MethodOneMsg@{
      argOne := 1;
      argTwo := 2;
      argThree := 3
    };
end;

module message_two_example;
  example_message_two : TemplateMsg :=
    MethodTwoMsg@{
      argOne := 1
    };
end;
